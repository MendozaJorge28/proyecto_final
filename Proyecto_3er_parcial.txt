import tkinter as tk
from tkinter import messagebox

class Persona:
    def __init__(self, nombre, edad, sexo, cedula, nacimiento, domicilio, telefono, correo, inicio):
        self.nombre = nombre
        self.edad = edad
        self.sexo = sexo
        self.cedula = cedula
        self.nacimiento = nacimiento
        self.domicilio = domicilio
        self.telefono = telefono
        self.correo = correo
        self.inicio = inicio

def iniciar_ventana():
    ventana = tk.Tk()
    ventana.title("Control de asistencias")
    ventana.geometry("500x400")
    ventana.config(bg="sky blue")

    tk.Label(ventana, text="Nombre", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Edad", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Sexo", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Cedula", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Nacimiento", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Domicilio", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Telefono", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Correo", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()
    tk.Label(ventana, text="Fecha de inicio", bg="sky blue").pack()
    nombre_entry = tk.Entry(ventana)
    nombre_entry.pack()


    ventana.mainloop()
iniciar_ventana()
